9. Treść procedury
function binarysearch (var a,b   : Extended;
                       mit       : Integer;
                       tol       : Extended;
                       var fatx  : Extended;
                       var it,st : Integer) : Extended;
var fa,fb,x : Extended;
begin
  try
    if mit<1
      then st:=1
      else begin
             fa:=f(a);
             fb:=f(b);
             if fa*fb>=0
               then st:=2
               else begin
                      st:=3;
                      it:=0;
                      repeat
                        it:=it+1;
                        x:=a+(b-a)/2;
                        fa:=f(x);
                        if (fa=0) or ((b-a)/2<tol)
                          then st:=0
                          else begin
                                 fa:=f(a);
                                 fb:=f(x);
                                 if fa*fb>0
                                   then a:=x
                                   else b:=x
                               end
                      until (it=mit) or (st<>3)
                    end
           end;
    if (st=0) or (st=3)
      then begin
             binarysearch:=x;
             fatx:=f(x)
           end
  except
    ShowMessage('Blad podczas wykonywania metody polowienia: nieprawidłowy zakres przedziału lub źle wczytana funkcja');
  end;
end;
